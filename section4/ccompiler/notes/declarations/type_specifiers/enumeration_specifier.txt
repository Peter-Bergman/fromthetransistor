         --------------------- Enumeration specifiers ------------------------------

         ----------- Syntax ------------

enum-specifier:
	enum identifier(opt) { enumerator-list }
	enum identifier(opt) { enumerator-list , }
	enum identifier
enumerator-list:
	enumerator
	enumerator-list , enumerator
enumerator:
	enumeration-constant
	enumeration-constant = constant-expression
